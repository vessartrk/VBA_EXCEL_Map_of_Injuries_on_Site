Sub Fac_Obs_ini()
'Dim FacxObsCol As Long, FacxObsRow As Long, FacLoop As Long
'Col = 116 - 125
'Row = 48 - 59
With sh_Map
    For FacLoop = 1 To ObservationsNum
        If FacxObsCol = 0 Then FacxObsCol = 116
        If FacxObsRow = 0 Then FacxObsRow = 48
        If FacxObsRow = 60 Then
            MsgBox "No more space for Facilities Observations"
            GoTo SkipFacLoop
        End If
        TopObsPos = .Range(FacxObsRow & ":" & FacxObsRow).Top
        LeftObsPos = .Cells(FacxObsRow, FacxObsCol).Left
        sh_Map.Shapes("ButtonMedGr").Duplicate.Name = "ShpObservation" & Format(NewShapeIDNum)
            With sh_Map.Shapes("ShpObservation" & Format(NewShapeIDNum))
                .Top = TopObsPos
                .Left = LeftObsPos
            FacxObsCol = FacxObsCol + 1
            If FacxObsCol >= 126 Then
                FacxObsCol = 116
                FacxObsRow = FacxObsRow + 1
            End If
        NewShapeIDNum = NewShapeIDNum + 1
        End With
        If FacLoop = ObservationsNum Then GoTo SkipFacLoop
    Next FacLoop
End With
SkipFacLoop:
MainObsLoop = MainObsLoop + 1
Call Observations_MainLoop_Month
End Sub

            With sh_Map.Shapes("ShpObservation" & Format(NewShapeIDNum))
                .Top = TopObsPos
                .Left = LeftObsPos
            SidxObsCol = SidxObsCol + 1
            If SidxObsCol >= 8 Then
                SidxObsCol = 6
                SidxObsRow = SidxObsRow + 1
            End If
        NewShapeIDNum = NewShapeIDNum + 1
        End With
        If SidLoop = ObservationsNum Then GoTo SkipSidLoopY
    Next SidLoop
End With
SkipSidLoopY:
MainObsLoop = MainObsLoop + 1
Call Observations_MainLoop_Year
End Sub
Sub Loc_Obs_ini_Year()
'Dim LocxObsCol As Long, LocxObsRow As Long, LocLoop As Long
'Col = 129 - 142
'Row = 12 - 15
With sh_Map
    For LocLoop = 1 To ObservationsNum
        If LocxObsCol = 0 Then LocxObsCol = 129
        If LocxObsRow = 0 Then LocxObsRow = 12
        If LocxObsRow = 16 Then
            MsgBox "No more space for Lockers Observations"
            GoTo SkipLocLoopY
        End If
        TopObsPos = .Range(LocxObsRow & ":" & LocxObsRow).Top
        LeftObsPos = .Cells(LocxObsRow, LocxObsCol).Left
        sh_Map.Shapes("ButtonMedGr").Duplicate.Name = "ShpObservation" & Format(NewShapeIDNum)
            With sh_Map.Shapes("ShpObservation" & Format(NewShapeIDNum))
                .Top = TopObsPos
                .Left = LeftObsPos
            LocxObsCol = LocxObsCol + 1
            If LocxObsCol >= 143 Then
                LocxObsCol = 129
                LocxObsRow = LocxObsRow + 1
            End If
        NewShapeIDNum = NewShapeIDNum + 1
        End With
        If LocLoop = ObservationsNum Then GoTo SkipLocLoopY
    Next LocLoop
End With
SkipLocLoopY:
MainObsLoop = MainObsLoop + 1
Call Observations_MainLoop_Year
End Sub
