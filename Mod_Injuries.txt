Sub FilterByMonth()

Dim LastRowFM As Long
Dim CalLoop As Long
InjuriesFilterByMonth = True

InjuriesFilterbyYear = False
Application.ScreenUpdating = False

With Sheet4
    
    .Range("AA5: AS999").Clear 'Clear all data in the database
    LastRowFM = .Range("R99999").End(xlUp).Row 'Determine Last Row
    
    For CalLoop = 1 To LastRowFM
   
        .Range("C4:S" & LastRowFM).AdvancedFilter xlFilterCopy, CriteriaRange:=.Range("Y4:Z5"), CopyToRange:=.Range("AA4:AQ4"), Unique:=True
    
    Next CalLoop
    
End With

Application.ScreenUpdating = True
Call FilterByType
End Sub

Sub FilterByYear()

InjuriesFilterbyYear = True
Dim LastRowFY As Long
Dim FilterByYearLoop As Long

Application.ScreenUpdating = False

End With

Application.ScreenUpdating = True
Call FilterByType
End Sub

Sub FilterByType()

Dim FilterTypeLoop As Long

Application.ScreenUpdating = False
' Check for the Type of the injury and set its value next to the advanced filter''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''
With Sheet4
    
    .Range("AR5:AS9999").Clear
    LastRowFilter = .Range("AA9999").End(xlUp).Row
    
    If LastRowFilter < 5 Then
        If InjuriesFilterbyYear = True Then
            MsgBox "Nothing for selected year"
            End
    
        ElseIf InjuriesFilterByMonth = True Then
            MsgBox "Nothing for selected Month"
            End
        End If
    End If
    
    For FilterTypeLoop = 5 To LastRowFilter
        Severity = .Cells(FilterTypeLoop, 35).Value
        FirstAid = .Cells(FilterTypeLoop, 34).Value
        
            If Severity = "LTI" Then
                .Cells(FilterTypeLoop, 44).Value = True
                .Cells(FilterTypeLoop, 45).Value = "Na"
            Else
                .Cells(FilterTypeLoop, 44).Value = False
                If FirstAid = "Yes" Then
                    .Cells(FilterTypeLoop, 45).Value = True
                Else
                    .Cells(FilterTypeLoop, 45).Value = False
                End If
            End If
    Next FilterTypeLoop
End With
Call Injuries_Initialization
End Sub

Private Sub Execute_Facilities()

    If FacxLTICol = 126 Then
        GoTo facLTIskip
    End If
' Assigning all LTI values
    If BolLti <> True Then
        GoTo facLTIskip
    End If

    If FacxLTICol = 0 Then
        FacxLTICol = 116
    End If
    If FacxLTIRow = 0 Then
        FacxLTIRow = 43
    End If
With sh_Map
        'TargCell = .Cells(xRow, xCol)
    TopPos = .Range(FacxLTIRow & ":" & FacxLTIRow).Top
    LeftPos = .Cells(FacxLTIRow, FacxLTICol).Left
    sh_Map.Shapes("ButtonLarge").Duplicate.Name = "ShpInjuryIsLTI" & Format(MainLoopCount)
    With sh_Map.Shapes("ShpInjuryIsLTI" & Format(MainLoopCount))
        .Top = TopPos
        .Left = LeftPos
    End With
    FacxLTICol = FacxLTICol + 2
    If FacxLTICol = 126 Then
        MsgBox "No More Space for Facilities LTI Injuries"
    End If
End With
    GoTo FacExeEnd
facLTIskip:
' Assigning all NON-LTI values
With sh_Map
    If FacxNonLTIRow = 48 Then
        GoTo FacExeEnd
    End If
    If FacxNonLTICol = 125 And FacxNonLTIRow = 47 Then  'AP is the Last Column
        MsgBox "No More Space for Facilities NON-LTI Injuries"
    End If
    If FacxNonLTICol = 0 Then
        FacxNonLTICol = 116
    End If
    If FacxNonLTIRow = 0 Then
        FacxNonLTIRow = 45
    End If
    
    TopPos = .Range(FacxNonLTIRow & ":" & FacxNonLTIRow).Top
    LeftPos = .Cells(FacxNonLTIRow, FacxNonLTICol).Left
    sh_Map.Shapes("ButtonMedRd").Duplicate.Name = "ShpInjuryNoLTI" & Format(MainLoopCount)
    With sh_Map.Shapes("ShpInjuryNoLTI" & Format(MainLoopCount))
        .Top = TopPos
        .Left = LeftPos
        If BolAid <> True Then
            .Fill.ForeColor.RGB = RGB(255, 191, 0)
        End If
    End With

    FacxNonLTICol = FacxNonLTICol + 1
  
    If FacxNonLTICol = 126 Then
        FacxNonLTIRow = FacxNonLTIRow + 1
        FacxNonLTICol = 116
    End If
End With
FacExeEnd:
MainLoopCount = MainLoopCount + 1
Call Main_Loop
End Sub
